import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# --- Step 1: Load and Prepare Weather Data ---
# Adjust the file path as needed
weather_df = pd.read_csv('Alberta weather 2017-2021 Daily.csv', parse_dates=['Date'])

# weather_df.head()

# 2. Rename relevent columns for convenience
weather_df.rename(columns={
    "Air Temp. Avg. (C)": "AvgTemp",
    "Precip. (mm)": "Precip",
    "Precip. Accumulated (mm)": "PrecipAccum",
    "Relative Humidity Avg. (%)": "Humidity",
    "Incoming Solar Rad. Total (MJ/m2)": "SolarRad",
    "Wind Speed 10 m Avg. (km/h)": "WindSpeed",
}, inplace=True)

#drop all irrelevant columns
weather_df = weather_df[['Date','AvgTemp','Precip','Humidity','SolarRad','WindSpeed']]

# Create year and month columns
weather_df['year'] = weather_df['Date'].dt.year
weather_df['month'] = weather_df['Date'].dt.month

#removing month october to frb because they are not relevant to honey bees life cycle or honey production as in these months there is No brood (typically), Bees cluster for warmth and Queen stops or greatly reduces egg-laying. so only considering from march to sept for analysis
weather_df = weather_df[weather_df["month"].between(3, 9)]

#add another column based on how close the temperature is to ideal temperature of 28 degree celsius.
weather_df["DiffFrom28"] = (weather_df["AvgTemp"] - 28).abs()

monthly_stats = weather_df.groupby(["year", "month"]).apply(lambda g: pd.Series({
    # Temperature
    "AvgTemp": g["AvgTemp"].mean(),
    
    # Threshold-based day counts (example: 25C)

    "Average_Diff_28": g["DiffFrom28"].mean(),
    "Days_Within_15Deg": (g["DiffFrom28"] <= 15).sum(),
    
    # Precipitation totals/averages
    "Total_Precip": g["Precip"].sum(),  # sum of daily precipitation
    
    # Humidity
    "Mean_Humidity": g["Humidity"].mean(),
    
    # Wind
    "Mean_WindSpeed": g["WindSpeed"].mean(),
    
    # Solar Radiation
    "Total_SolarRad": g["SolarRad"].sum()
}),include_groups = False, ).reset_index()

# 5. Inspect the resulting monthly_stats

# 1. Map month -> bee_stage
bee_stage_map = {
    3: 'Larva',
    4: 'Pupa',
    5: 'Pupa',
    6: 'Adult',
    7: 'Adult',
    8: 'Adult',
    9: 'Adult',
    10: 'Larva'
}
weather_df['bee_stage'] = weather_df['month'].map(bee_stage_map)

# 2. Group by year & stage, aggregate variables
stage_stats = weather_df.groupby(['year','bee_stage'], as_index=False).agg(
    AvgTemp=('AvgTemp', 'mean'),
    Precip=('Precip', 'sum'),
    Humidity=('Humidity', 'mean'),
    SolarRad=('SolarRad', 'mean'),
    WindSpeed=('WindSpeed', 'mean'),
    Average_Diff_28=('DiffFrom28', 'mean'),
    Days_Within_15Deg=('DiffFrom28', lambda x: (x <= 15).sum()),
)


# 3. Pivot so each (bee_stage, variable) is a separate column
stage_pivot = stage_stats.pivot_table(
    index='year',
    columns='bee_stage', 
    values=['AvgTemp','Precip','Humidity','SolarRad','WindSpeed','Average_Diff_28','Days_Within_15Deg']
).reset_index()

# 4. Flatten the multi-level columns
stage_pivot.columns = [
    f"{var}_{stage}" if stage else var  # combine variable + stage into a single string
    for var, stage in stage_pivot.columns
]
# After this, you'll have columns like "Precip_Larva", "Humidity_Adult", etc., plus "year".

# 5. Merge with annual honey production
honey_df = pd.DataFrame({
    'year': [2017, 2018, 2019, 2020, 2021],
    'HoneyProduction': [40576000, 38564000, 31418000, 30735000, 34925000]
})
df_merged = pd.merge(stage_pivot, honey_df, on='year', how='inner')
df_merged = df_merged.drop(columns='year')
# 6. Correlation with annual honey production
corr_series = df_merged.corr(numeric_only=True)['HoneyProduction'].drop('HoneyProduction')
corr_sorted = corr_series.sort_values(ascending=False)
