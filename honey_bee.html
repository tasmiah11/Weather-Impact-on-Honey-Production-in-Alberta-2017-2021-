# 2. Rename relevent columns for convenience
df.rename(columns={
    "Air Temp. Avg. (C)": "AvgTemp",
    "Precip. (mm)": "Precip",
    "Precip. Accumulated (mm)": "PrecipAccum",
    "Relative Humidity Avg. (%)": "Humidity",
    "Incoming Solar Rad. Total (MJ/m2)": "SolarRad",
    "Wind Speed 10 m Avg. (km/h)": "WindSpeed",
}, inplace=True)

#drop all irrelevant columns
df = df[['Date','AvgTemp','Precip','PrecipAccum','Humidity','SolarRad','WindSpeed']]

# Create Year and Month columns
df['Year'] = df['Date'].dt.year
df['Month'] = df['Date'].dt.month
In[2]:
#removing month october to frb because they are not relevant to honey bees life cycle or honey production as in these months there is No brood (typically), Bees cluster for warmth and Queen stops or greatly reduces egg-laying. so only considering from march to sept for analysis
df = df[df["Month"].between(3, 9)]
In[3]:
#add another column based on how close the temperature is to ideal temperature of 28 degree celsius.
df["DiffFrom28"] = (df["AvgTemp"] - 28).abs()
In[4]:
summary = df.groupby(["Year", "Month"]).apply(lambda g: pd.Series({
    # Temperature
    "Mean_of_AvgTemp": g["AvgTemp"].mean(),
    
    # Threshold-based day counts (example: 25C)

    "Average_Diff_28": g["DiffFrom28"].mean(),
    "Days_Within_15Deg": (g["DiffFrom28"] <= 15).sum(),
    
    # Precipitation totals/averages
    "Total_Precip": g["Precip"].sum(),  # sum of daily precipitation
    "Monthly_Accum_Precip": g["PrecipAccum"].max() - g["PrecipAccum"].min() if not g["PrecipAccum"].isna().all() else None,
    
    # Humidity
    "Mean_Humidity": g["Humidity"].mean(),
    
    # Wind
    "Mean_WindSpeed": g["WindSpeed"].mean(),
    
    # Solar Radiation
    "Total_SolarRad": g["SolarRad"].sum()
}),include_groups = False, ).reset_index()

import matplotlib.pyplot as plt
import seaborn as sns

stages = ['Larva', 'Pupa', 'Adult']
fig, axes = plt.subplots(1, len(stages), figsize=(18, 5))
for i, stage in enumerate(stages):
    if stage in combined.columns:
        sns.regplot(x=combined[stage], y=combined['Production'], ax=axes[i])
        axes[i].set_title(f'Production vs. Avg {stage} Humidity')
        axes[i].set_xlabel(f'Avg {stage} Humidity')
        axes[i].set_ylabel('Honey Production')
plt.tight_layout()
plt.show()

# Also plot overall average humidity versus production.
plt.figure(figsize=(8,6))
sns.regplot(x=combined['overall_humidity'], y=combined['Production'])
plt.title('Production vs. Overall Avg Humidity (Active Months)')
plt.xlabel('Overall Avg Humidity')
plt.ylabel('Honey Production')
plt.show()
